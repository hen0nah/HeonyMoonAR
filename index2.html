async function loadCSVData() { // 1. 위에서 만든 구글 스프레드시트 CSV URL을 여기에 넣습니다. const csvUrl = 'https://docs.google.com/spreadsheets/d/12345abcdeFGHIJKLMNOPQRS_TUV_WXYZ/export?format=csv'; try { const response = await fetch(csvUrl); const csvText = await response.text(); // 2. CSV 데이터를 줄 단위로 나눕니다. const rows = csvText.trim().split('\n'); // 3. HTML 테이블을 생성합니다. const table = document.createElement('table'); // 4. 헤더를 만듭니다. const headerRow = document.createElement('tr'); const headers = rows[0].split(','); headers.forEach(header => { const th = document.createElement('th'); th.textContent = header.trim(); headerRow.appendChild(th); }); table.appendChild(headerRow); // 5. 데이터를 만듭니다. for (let i = 1; i < rows.length; i++) { const dataRow = document.createElement('tr'); const data = rows[i].split(','); data.forEach(cell => { const td = document.createElement('td'); td.textContent = cell.trim(); dataRow.appendChild(td); }); table.appendChild(dataRow); } // 6. 생성된 테이블을 HTML 문서에 추가합니다. document.body.appendChild(table); } catch (error) { console.error('데이터를 불러오는 중 오류 발생:', error); } } // 함수 호출 loadCSVData();
